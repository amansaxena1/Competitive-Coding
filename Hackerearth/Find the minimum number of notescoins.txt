#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <bits/stdc++.h> 
#include <algorithm>
using namespace std;


int main() {
    long long int x , n , i , xx , sum = 0;
    cin>>x>>n;
    vector<long long int> v;
    
    for(i = 0 ; i < n ; i++){
        cin>>xx;
        v.push_back(xx);
    }
    
    sort(v.begin(), v.end(), greater<int>()); 
    
    for(i = 0 ; i < n ; i++){
        sum = sum + (x/v[i]);
        x = x % v[i];
    }
    cout<<sum;
    return 0;
}
